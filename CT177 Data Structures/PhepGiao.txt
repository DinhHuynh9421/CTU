	E2 = endList(B);
	ElementType X;
	makenullList(C);

}

void Nhap(List *pL){
	int n, i;
	ElementType X;

	for(i=1; i<=n; i++){
		scanf("%d", &X);
		if(locate(X, *pL)== endList(*pL))
		  insertList(X,endList(*pL), pL);
		  
	}
}
